package jcomb.core.parameter;


public class IntFromTo implements Domain {

  private int from;
  private int to;
  private int step;



  public IntFromTo(int from, int to) {
    this(from, to, 1);
  }



  public IntFromTo(int from, int to, int step) {
    this.from = from;
    this.to = to;
    this.step = step;
  }



  @Override
  public Object getValueAt(int index) {
    return from + index*step;
  }



  @Override
  public int getSize() {
    return (to - from + 1)/step;
  }

}
