package jcomb.core.constraint;


/**
 * A constraint handler can check if a (partial) test is or can satisfy all constraints in the constraint handler.
 * @author Noah Zuch
 *
 */
public interface ConstraintHandler {

  
  public boolean isSatisfiable(int[] test);

  public static ConstraintHandler getDefaultConstraintHandler() {
    return new ConstraintHandler() {
      
      @Override
      public boolean isSatisfiable( int[] test) {
       return true;
      }
    };
  }
}
