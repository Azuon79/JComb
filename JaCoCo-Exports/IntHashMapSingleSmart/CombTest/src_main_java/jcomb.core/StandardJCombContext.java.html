package jcomb.core;

import java.util.ArrayList;
import java.util.Collection;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.stream.Collectors;
import jcomb.core.constraint.Constraint;
import jcomb.core.parameter.Domain;

/**
 * The default implementation of an JCombContext.
 * 
 * @author Noah Zuch
 *
 */
public class StandardJCombContext implements JCombContext {

  private List<Domain> parameters;
  private List<InstanceDependent> instanceDependant;
  private Map<Integer, Constraint> constraints;
  private Map<Integer, Domain> seeds;



  public StandardJCombContext() {
    parameters = new ArrayList<>();
    constraints = new HashMap<>();
    seeds = new HashMap<>();
  }



  @Override
  public Map<Integer, Constraint> getConstraints() {
    return constraints;
  }



  @Override
  public Domain getParameter(int index) {
    return parameters.get(index);
  }



  @Override
  public int getParameterCount() {
    return parameters.size();
  }



  public void setAlsoIncludes(Map<Integer, Domain> alsoIncludes) {
    // this.alsoIncludes = alsoIncludes;
  }



  public void setConstraints(Map<Integer, Constraint> constraints) {
    this.constraints = constraints;
  }



  public Map<Integer, Domain> getAlsoIncludesMap() {
    return seeds;
  }



  public Map<Integer, Constraint> getConstraintsMap() {
    return constraints;
  }



  @Override
  public void initializeForInstance(Object currentTestInstance) {
    instanceDependant.forEach(obj -> obj.initializeForInstance(currentTestInstance));
  }



  public void setParameters(List<Domain> parameters) {
    this.parameters = parameters;
    this.instanceDependant = parameters.stream()
        .filter(obj -> obj instanceof InstanceDependent)
        .map(obj -> (InstanceDependent) obj).collect(Collectors.toList());
  }

  @Override
  public Collection<int[]> getSeeds() {
    return null;// seeds.values();
  }

}
