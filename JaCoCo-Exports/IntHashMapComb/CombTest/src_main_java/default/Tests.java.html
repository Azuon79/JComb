  private static String toBinaryString(long v) {
    StringBuilder binary = new StringBuilder();
    long mask = 1l << 63;
    while (mask != 0) {
      if ((mask & v) == 0) {
        binary.append(0);
      } else
        binary.append(1);
      mask = mask >>> 1;
    }
    return binary.toString();
  }



  private static String toBinaryString(int[] paramComb, int size) {
    StringBuilder builder = new StringBuilder();
    int paramCombIndex = 0;
    for (int i = 0; i < size; i++) {
      if (paramComb[paramCombIndex] == i) {
        builder.insert(0, '1');
        if (paramCombIndex < paramComb.length - 1) {
          paramCombIndex++;
        }
      } else {
        builder.insert(0, '0');
      }
    }
    return builder.toString();
  }

  private static int getIndexFromParamComb(int[] comb, int paramCount) {

  private static void getNextParamComb(int[] currentComb) {
    int lastParam = currentComb[0];
    int adjacentParamCount = 0;
    for (int i = 1; i < currentComb.length; i++) {
      if (currentComb[i] == lastParam + 1) {
        lastParam = currentComb[i];
        adjacentParamCount++;
        if (i == currentComb.length - 2) {
          currentComb[i] += 1;
          break;
        }
      } else {
        currentComb[i - 1] += 1;
        break;
      }
    }
    for (int i = 0; i < adjacentParamCount; i++) {
